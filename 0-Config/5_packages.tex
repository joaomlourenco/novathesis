% ShadedTabular: a tabular with a shaded header row and zebra-striped body.
% Uses colortbl (already loaded by novathesis) — no extra packages needed.
% The header row receives headgray; body rows alternate rowgray / white.
\newenvironment{ShadedTabular}[1]{%
  \renewcommand{\arraystretch}{1.3}%
  \rowcolors{2}{rowgray}{white}%    % body rows: zebra stripes starting at row 2
  \tabular{#1}%
  \rowcolor{headgray}%              % first (header) row background
}{%
  \endtabular
}
