\typeout{NT FILE StyFiles/fix-babel-expl3.tex}%

\ExplSyntaxOn

%% INITIALIZATION & STORAGE
\seq_new:N \g_nt_babel_tracked_langs_seq

%% LOCALIZED KEY CONFIGURATION (\ntlangsetup)
\NewDocumentCommand{\ntlangsetup}{ s O{name} m }
  {
    \nt_babel_parse_setup:nnnn { #1 } { #2 } #3 \q_stop
  }

\cs_new_protected:Npn \nt_babel_parse_setup:nnnn #1 #2 #3 / #4 = #5 \q_stop
  {
    \nt_babel_store_key:nnnnn { #1 } { #2 } { #3 } { #4 } { #5 }
  }

\cs_new_protected:Nn \nt_babel_store_key:nnnnn
  {
    \prop_if_exist:cF { g_nt_babel_#3_prop }
      {
        \prop_new:c { g_nt_babel_#3_prop }
        \seq_gput_right:Nn \g_nt_babel_tracked_langs_seq { #3 }
      }

    \bool_if:nTF { #1 }
      { \prop_gput_if_new:cnn { g_nt_babel_#3_prop } { #4 #2 } { #5 } }
      { \prop_gput:cnn { g_nt_babel_#3_prop } { #4 #2 } { #5 } }
  }

%% ENGINE: APPLYING LOCALIZED CAPTIONS
\cs_new_protected:Nn \nt_babel_apply_all_captions:
  {
    \cs_if_exist:NT \optionlistdo
      {
        \optionlistdo { /@nt/lang/loaded }
          {
            \prop_if_exist:cT { g_nt_babel_ ##1 _prop }
              {
                \prop_map_inline:cn { g_nt_babel_ ##1 _prop }
                  {
                    \setlocalecaption { \GetLangLong(##1) } { ####1 } { ####2 }
                  }
              }
          }
      }
  }

% Hook into the "very end" of the preamble.
% We check for \hook_gadd:nnn (the Hook Manager) 
% and fallback to \AtBeginDocument if it's an old engine.
\cs_if_exist:NTF \hook_gadd:nnn
  { 
    \hook_gadd:nnn { begindocument / before } { \ExplSyntaxOn \nt_babel_apply_all_captions: \ExplSyntaxOff } 
  }
  { 
    \AtBeginDocument { \ExplSyntaxOn \nt_babel_apply_all_captions: \ExplSyntaxOff } 
  }

%% COMPATIBILITY HELPERS
% For compatibility with Babel version < 3.51
\cs_set_protected:Npn \setlocalecaption #1 #2 #3
  {
    \expandafter\addto\csname captions#1\endcsname { \cs_set:cpn {#2} {#3} }
  }

\ExplSyntaxOff

\addto\extrasportuguese{%
  \defineshorthand{"-}{\babelhyphen{repeat}}%
