%%=======================================================================
%% CUSTOM LIST ENVIRONMENTS (emulating paralist)
%%=======================================================================

\newenvironment{asparaenum}[1][]
  {\begin{enumerate}[#1, wide, itemsep=0pt, parsep=\parskip, topsep=0pt, partopsep=0pt]}
  {\end{enumerate}}

\newenvironment{inparaenum}[1][]
  {\begin{enumerate*}[#1]}
  {\end{enumerate*}}

\newenvironment{compactitem}[1][]
  {\begin{itemize}[#1, noitemsep, topsep=0pt, parsep=0pt, partopsep=0pt]}
  {\end{itemize}}

\newenvironment{compactenum}[1][]
  {\begin{enumerate}[#1, noitemsep, topsep=0pt, parsep=0pt, partopsep=0pt]}
  {\end{enumerate}}
